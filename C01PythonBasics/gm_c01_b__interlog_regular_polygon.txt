## --- section_c: area of n-th regular polygon ---
>>> rr = 2  # radius of circumscribed circle
>>> nn = 3  # number of sides
>>> th = 2 * pi / nn
>>> ss_tri = 1 /2 * (2 * rr * sin(th/2)) * (rr * cos(th/2))
>>> ss = nn * ss_tri
>>> ss
5.196152422706633
